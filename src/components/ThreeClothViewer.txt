// -------------------- HOOK: Generic texture loader --------------------
const __textureCache = new Map();

function useTextureLoader(url, options = {}) {
    const { gl } = useThree();
    const wrap = options.wrap || THREE.ClampToEdgeWrapping;
    const encoding = options.encoding || 'sRGB';

    const texture = useMemo(() => {
        if (!url) return null;
        const repeat = options.repeat || [1, 1];

        // If we have a cached base texture for this URL, clone it and apply requested params.
        if (__textureCache.has(url)) {
            try {
                const base = __textureCache.get(url);
                const tex = base.clone();
                tex._loaded = !!base._loaded;
                tex.flipY = false;
                tex.wrapS = tex.wrapT = wrap;
                tex.repeat.set(...repeat);
                tex.generateMipmaps = false;
                tex.minFilter = THREE.LinearFilter;
                tex.magFilter = THREE.LinearFilter;
                tex.anisotropy = Math.min(4, base.anisotropy || 2);
                try {
                    if (encoding === 'sRGB') {
                        if ('SRGBColorSpace' in THREE) tex.colorSpace = THREE.SRGBColorSpace;
                    } else if (encoding === 'linear') {
                        if ('LinearSRGBColorSpace' in THREE) tex.colorSpace = THREE.LinearSRGBColorSpace;
                    }
                } catch (e) {}
                return tex;
            } catch (e) {
                // fallback to fresh load below
            }
        }

        // First time load: create and cache the base texture
        const loader = new THREE.TextureLoader();
        const tex = loader.load(
            url,
            (image) => {
                tex._loaded = true;
                try {
                    if (tex.userData && Array.isArray(tex.userData._onLoadCallbacks)) {
                        tex.userData._onLoadCallbacks.forEach(fn => { try { fn(); } catch(e){} });
                        tex.userData._onLoadCallbacks.length = 0;
                    }
                } catch (e) {}
                // Cache the loaded texture for future clones
                try { __textureCache.set(url, tex); } catch (e) {}
            },
            undefined,
            (err) => console.error('Error loading texture:', err)
        );

        tex.flipY = false;
        tex.wrapS = tex.wrapT = wrap;
        tex.repeat.set(...repeat);
        try {
            const maxAniso = (gl && gl.capabilities && typeof gl.capabilities.getMaxAnisotropy === 'function')
                ? gl.capabilities.getMaxAnisotropy()
                : 2;
            tex.anisotropy = Math.min(4, maxAniso || 2);
        } catch (e) {
            tex.anisotropy = 2;
        }
        tex.generateMipmaps = false;
        tex.minFilter = THREE.LinearFilter;
        tex.magFilter = THREE.LinearFilter;
        try {
            if (encoding === 'sRGB') {
                if ('SRGBColorSpace' in THREE) tex.colorSpace = THREE.SRGBColorSpace;
            } else if (encoding === 'linear') {
                if ('LinearSRGBColorSpace' in THREE) tex.colorSpace = THREE.LinearSRGBColorSpace;
            }
        } catch (e) {}

        return tex;
    }, [url, options.repeat, wrap, encoding, gl]);

    return texture;
}

// -------------------- ESCENA PRINCIPAL (Materiales y Autogiro) --------------------
function Scene({ BaseModel, selectedPattern, baseColor, ClothingComponent, patternOptions, logoColor, logoUrl, modelGroupRef, rotationGoalRef, isAutoRotate, clothingCacheRef, isInteracting }) { // eslint-disable-line no-unused-vars
    // 💡 CORRECCIÓN: Definimos la ref de la escena al principio del componente.
    const sceneRef = useRef();
    // La animación ahora debe ser manejada de forma diferente si es necesaria.
    // Por ahora, la omitimos ya que el modelo base parece estático.
    const mixer = useRef(null);

    const clothMaterialRef = useRef();
    const baseMaterialRef = useRef();
    const { invalidate } = useThree();

    // Load lycra base map (user provided PNG in src/assets)
    const lycraBase = useTextureLoader(lycraImg, {
        wrap: THREE.RepeatWrapping,
        repeat: [2, 2],
        encoding: 'sRGB'
    });

        // Removed procedural roughness generation to save CPU at startup.

    // Permanently ignore any pattern that contains 'patron1'
    const effectivePattern = selectedPattern && selectedPattern.includes('patron1') ? null : selectedPattern;

    // Load pattern (if any). We still create the Texture object immediately
    // so it can be attached synchronously to materials (even if image not yet loaded).
    const patternTexture = useTextureLoader(effectivePattern, { 
        wrap: THREE.RepeatWrapping, 
        repeat: [patternOptions.repeatX, patternOptions.repeatY] 
    });

    // Use a lower-anisotropy texture load to reduce memory/processing on load
    const logoTexture = useTextureLoader(logoUrl, { 
        wrap: THREE.ClampToEdgeWrapping, 
        repeat: [1, 1] 
    });

    // 🔑 LÓGICA DE MATERIALES CENTRALIZADA
    useEffect(() => {
        const mat = clothMaterialRef.current;
        if (!mat) return;

        const hasPattern = !!patternTexture;
        const hasLoadedPattern = patternTexture && patternTexture._loaded;

        if (hasPattern && hasLoadedPattern) {
            mat.map = patternTexture;
            mat.color.set(baseColor.substring(0, 7));
        } else {
            mat.map = lycraBase || null;
            mat.color.set(baseColor.substring(0, 7));
        }

        if (hasPattern && !hasLoadedPattern) {
            patternTexture.userData = patternTexture.userData || {};
            patternTexture.userData._onLoadCallbacks = patternTexture.userData._onLoadCallbacks || [];
            patternTexture.userData._onLoadCallbacks.push(() => {
                if (clothMaterialRef.current) {
                    clothMaterialRef.current.map = patternTexture;
                    clothMaterialRef.current.color.set(baseColor.substring(0, 7));
                    clothMaterialRef.current.needsUpdate = true;
                    invalidate();
                }
            });
        }

        mat.needsUpdate = true;
        invalidate();

    }, [baseColor, patternTexture, lycraBase, invalidate]);

    // Efecto para actualizar el color del logo
    useEffect(() => {
        sceneRef.current?.traverse(child => {
            if (child.isMesh) {
                if (LOGO_MESH_NAMES.includes(child.name)) {
                    child.material = new THREE.MeshStandardMaterial({
                        color: new THREE.Color(logoColor),
                        map: logoTexture,
                        transparent: true,
                        alphaTest: 0.1,
                        side: THREE.DoubleSide,
                        roughness: 0.45,
                        metalness: 0.0,
                        envMapIntensity: 0.35
                    });
                }
                // El material de la ropa se aplica en el componente ModelRenderer
            }
        });
        invalidate();
    }, [logoColor, logoTexture, invalidate]);

    // Efecto para despachar el evento 'clothing-loaded' cuando las texturas están listas
    useEffect(() => {
        if (ClothingComponent) {
            window.dispatchEvent(new Event('clothing-loading'));
        }
    }, [ClothingComponent]);

    // Efecto para despachar el evento 'clothing-loaded' cuando las texturas están listas
    useEffect(() => {
        // Dispatch event when ready
        const texturesToWait = [patternTexture, lycraBase, logoTexture].filter(Boolean);
        const check = () => {
            if (texturesToWait.every(t => t._loaded)) {
                if (ClothingComponent) window.dispatchEvent(new Event('clothing-loaded'));
                invalidate();
            }
        };
        check();
        texturesToWait.forEach(t => {
            if (t && !t._loaded) {
                t.userData = t.userData || {};
                t.userData._onLoadCallbacks = t.userData._onLoadCallbacks || [];
                t.userData._onLoadCallbacks.push(check);
            }
        });
    }, [ClothingComponent, invalidate, patternTexture, lycraBase, logoTexture]);

 // ... Animaciones y retornos (mantenidos)
    // NOTE: We no longer apply clothing-style materials to the base `scene` here.
    // Clothing materials are applied inside `ExtraClothing` to the cloned clothing GLB.

    useEffect(() => {
            // Dispatch an event once the base model and its textures are available (only once)
            // We can simplify this: assume the base model is ready quickly.
            // The clothing loader is more important.
            const to = setTimeout(() => window.dispatchEvent(new Event('base-model-loaded')), 500);
            return () => clearTimeout(to);
    }, []);

  // 🔑 ROTACIÓN: Auto-rotate y objetivos de rotación suaves
  // Replace continuous useFrame with a demand-based RAF loop that only renders when necessary.
  useEffect(() => {
      let rafId = null;
      let last = performance.now();
      function loop(now) {
          const delta = (now - last) / 1000;
          last = now; 
          try { mixer.current?.update(delta); } catch (e) {}
          if (!modelGroupRef?.current) return;
          const cur = modelGroupRef.current.rotation.y;
          let shouldInvalidate = false;
          if (rotationGoalRef?.current !== null && rotationGoalRef?.current !== undefined) {
              const next = THREE.MathUtils.lerp(cur, rotationGoalRef.current, 0.12);
              modelGroupRef.current.rotation.y = next;
              if (Math.abs(next - rotationGoalRef.current) < 0.001) {
                  rotationGoalRef.current = null;
              }
              if (Math.abs(next - cur) > 1e-5) shouldInvalidate = true;
          } else if (isAutoRotate) {
              const next = cur + delta * 0.5; // auto-rotate speed
              modelGroupRef.current.rotation.y = next;
              if (Math.abs(next - cur) > 1e-5) shouldInvalidate = true;
          }

          // If interacting but not auto-rotating, still request invalidation so the scene updates
          if (isInteracting && !isAutoRotate) {
              shouldInvalidate = true;
          }

          if (shouldInvalidate) invalidate();

          rafId = requestAnimationFrame(loop);
      }

      // Start the loop when auto-rotating, interacting, or there's an active rotation goal
      if (isAutoRotate || isInteracting || (rotationGoalRef && rotationGoalRef.current !== null)) {
          rafId = requestAnimationFrame(loop);
      }

      return () => { if (rafId) cancelAnimationFrame(rafId); };
    }, [isAutoRotate, isInteracting, invalidate, modelGroupRef, rotationGoalRef]);

  // Mantener listeners ligeros para debug (list-meshes y toggle-cloth)
  useEffect(() => {
    function onListMeshes() {
        const names = [];
        if (sceneRef.current) {
            sceneRef.current.traverse(child => { if (child.isMesh) names.push(child.name || '<mesh>'); });
        }
        console.log('3D Scene meshes:', names);
    }
    window.addEventListener('list-meshes', onListMeshes);
    return () => window.removeEventListener('list-meshes', onListMeshes);
  }, []);

  return (
    <>
      {/* 🔑 AUTOGIRO: El modelo está envuelto en un grupo para rotación */}
      <meshStandardMaterial
        ref={clothMaterialRef}
        side={THREE.DoubleSide}
        roughness={0.6}
        metalness={0.03}
        envMapIntensity={0.35}
      />
      <meshStandardMaterial
        ref={baseMaterialRef}
        color="#000000"
        metalness={1}
        roughness={0.2}
        envMapIntensity={0.5}
      />
      <group ref={modelGroupRef}> 
        <group ref={sceneRef}>
            {/* Renderizador para el modelo base */}
            <ModelRenderer
                key="baseModel"
                ModelComponent={BaseModel}
                sharedMaterial={baseMaterialRef.current}
                isBase={true}
            />

            {ClothingComponent && (
                <Suspense fallback={null}>
                    {/* Renderizador para la prenda seleccionada */}
                    <ModelRenderer
                        key={ClothingComponent.name}
                        ModelComponent={ClothingComponent}
                        sharedMaterial={clothMaterialRef.current}
                        isBase={false}
                    />
                </Suspense>
            )}
        </group>
      </group>
 </>
 );
}

/**
 * Este componente renderiza un modelo de gltfjsx y aplica el material compartido
 * a todas sus mallas, excepto a las que son logos.
 */
function ModelRenderer({ ModelComponent, sharedMaterial, isBase }) {
    const groupRef = useRef();

    // Add subtle movement animation for clothing
    useFrame((state) => {
        if (!isBase && groupRef.current) {
            // Subtle oscillation animation for clothing
            const time = state.clock.elapsedTime;
            groupRef.current.position.y = Math.sin(time * 0.5) * 0.005; // Very subtle vertical movement
            groupRef.current.rotation.z = Math.sin(time * 0.3) * 0.01; // Very subtle rotation
        }
    });

    useEffect(() => {
        if (!sharedMaterial || !groupRef.current) return;
        // Aplica el material compartido a las mallas del modelo
        groupRef.current.traverse(child => {
            if (child.isMesh && !LOGO_MESH_NAMES.includes(child.name)) {
                child.material = sharedMaterial;
                child.castShadow = true;
                child.receiveShadow = true;
            }
        });
    }, [sharedMaterial]);

    // Usamos React.forwardRef en el componente del modelo para obtener la ref del grupo
    return <ModelComponent ref={groupRef} />;
}

  const [isLoading, setIsLoading] = useState(true);

    const [localPattern, setLocalPattern] = useState(selectedPattern || null);
    // Controles de rotación centrales (model-only rotation)
    const modelGroupRef = useRef();
    const rotationGoalRef = useRef(null);
    const [isAutoRotate, setIsAutoRotate] = useState(true);
    const [isInteracting, setIsInteracting] = useState(false);
    const containerRef = useRef();
    const clothingCacheRef = useRef(new Map());

    // Listen for global pattern events (Catalog dispatches 'apply-pattern')
    useEffect(() => {
        function onApply(e) {
            const url = e?.detail?.url;
            if (url) setLocalPattern(url);
        }
        window.addEventListener('apply-pattern', onApply);
        return () => window.removeEventListener('apply-pattern', onApply);
    }, []);

    // Drag-to-rotate (model-only). Attach pointer handlers to containerRef
    useEffect(() => {
        const el = containerRef.current;
        if (!el) return;
        let dragging = false;
        let startX = 0;
        let startRot = 0;
        const sensitivity = 0.01; // Increased for more responsive feel

        function onPointerDown(e) {
            dragging = true;
            startX = e.clientX || (e.touches && e.touches[0]?.clientX) || 0;
            startRot = modelGroupRef.current?.rotation.y || 0;
            setIsAutoRotate(false);
            setIsInteracting(true);
            if (e.pointerId) el.setPointerCapture?.(e.pointerId);
        }

        function onPointerMove(e) {
            if (!dragging) return;
            const clientX = e.clientX || (e.touches && e.touches[0]?.clientX) || 0;
            const dx = clientX - startX;
            // Directly set the rotation goal for immediate response
            rotationGoalRef.current = startRot + dx * sensitivity;
        }

        function onPointerUp(e) {
            dragging = false;
            setIsInteracting(false);
            try { window.dispatchEvent(new Event('interaction-end')); } catch (err) {}
            if (e.pointerId) el.releasePointerCapture?.(e.pointerId);
        }

        el.addEventListener('pointerdown', onPointerDown);
        window.addEventListener('pointermove', onPointerMove);
        window.addEventListener('pointerup', onPointerUp);

        return () => {
            el.removeEventListener('pointerdown', onPointerDown);
            window.removeEventListener('pointermove', onPointerMove);
            window.removeEventListener('pointerup', onPointerUp);
        };
    }, []);

    // Global events for rotate buttons
    useEffect(() => {
        function onRotateLeft() {
            const cur = modelGroupRef.current?.rotation.y || 0;
            rotationGoalRef.current = cur + Math.PI / 6;
            setIsAutoRotate(false);
        }
        function onRotateRight() {
            const cur = modelGroupRef.current?.rotation.y || 0;
            rotationGoalRef.current = cur - Math.PI / 6;
            setIsAutoRotate(false);
        }
        function onSetRotation(e) {
            const angle = e?.detail?.angle;
            if (typeof angle === 'number') {
                rotationGoalRef.current = angle;
                setIsAutoRotate(false);
            }
        }
        function onToggleAutospin() {
            setIsAutoRotate(v => !v);
        }

        window.addEventListener('rotate-left', onRotateLeft);
        window.addEventListener('rotate-right', onRotateRight);
        window.addEventListener('set-rotation', onSetRotation);
        window.addEventListener('toggle-autospin', onToggleAutospin);

        return () => {
            window.removeEventListener('rotate-left', onRotateLeft);
            window.removeEventListener('rotate-right', onRotateRight);
            window.removeEventListener('set-rotation', onSetRotation);
            window.removeEventListener('toggle-autospin', onToggleAutospin);
        };
    }, []);

        // Track readiness of base model and clothing so loader stays until both ready
        const readyRef = useRef({ base: false, cloth: false });
        useEffect(() => {
            function onBase() {
                readyRef.current.base = true;
                if (readyRef.current.base && readyRef.current.cloth) setIsLoading(false);
            }
            function onCloth() {
                readyRef.current.cloth = true;
                if (readyRef.current.base && readyRef.current.cloth) setIsLoading(false);
            }
            // fallback: don't stay forever
            const to = setTimeout(() => setIsLoading(false), 4000);
            window.addEventListener('base-model-loaded', onBase);
            window.addEventListener('clothing-loaded', onCloth);
            return () => {
                clearTimeout(to);
                window.removeEventListener('base-model-loaded', onBase);
                window.removeEventListener('clothing-loaded', onCloth);
            };
        }, []);

        return (
            <div ref={containerRef} style={{ width: '100%', height: '100%', minHeight: 420, position: 'relative' }}>
            {isLoading && (
                <div style={{
                    position: 'absolute', top: 0, left: 0, right: 0, bottom: 0,
                    backgroundColor: 'rgba(0,0,0,0.04)',
                    display: 'flex', justifyContent: 'center', alignItems: 'center',
                    color: '#fff', zIndex: 20,
                    pointerEvents: 'none' // don't block interactions while showing loader
                }}>
                    <div style={{ display: 'flex', alignItems: 'center', gap: 12, background: 'rgba(0,0,0,0.45)', padding: '10px 16px', borderRadius: 10, boxShadow: '0 6px 22px rgba(0,0,0,0.45)' }}>
                        <svg width="28" height="28" viewBox="0 0 50 50" style={{ display: 'block' }}>
                            <path fill="none" stroke="#fff" strokeWidth="4" strokeLinecap="round" d="M25 5 A20 20 0 0 1 45 25" strokeDasharray="31.4 31.4">
                                <animateTransform attributeName="transform" type="rotate" from="0 25 25" to="360 25 25" dur="0.9s" repeatCount="indefinite" />
                            </path>
                        </svg>
                        <div style={{ color: '#fff', fontSize: 14, fontWeight: 600 }}>Cargando 3D — optimizando experiencia...</div>

                </div>
            )}
    <Canvas

        camera={{ position: [0, 1.9, 2.5], fov: 45 }}
        onCreated={({ gl }) => {
            // Clamp pixel ratio to avoid excessive render cost on high-DPI devices
            try { gl.setPixelRatio(Math.min(window.devicePixelRatio || 1, 1.25)); } catch (e) {}
        }}
    >
        {/* Environment removed to avoid loading issues */}
